Create PROCEDURE [dbo].[GetNewList] 
 @Page INT =1,
 @PageSize INT = 10,
 @SearchTerm VARCHAR(255)=''
AS
BEGIN
SET NOCOUNT ON;
SELECT
    Title,Url FROM Story WHERE
    (@SearchTerm IS NULL OR Title LIKE '%' + @SearchTerm + '%')
ORDER BY CreatedAt DESC
OFFSET
    (@Page - 1) * @PageSize ROWS
FETCH NEXT
    @PageSize ROWS ONLY;

END
